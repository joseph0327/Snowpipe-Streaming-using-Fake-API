USE ROLE ACCOUNTADMIN

-- Create role
CREATE OR REPLACE ROLE HEALTH_ROLE;

SHOW ROLES;

-- Create the Warehouse
CREATE OR REPLACE WAREHOUSE HEALTH_WH
    WAREHOUSE_SIZE = 'XSMALL'
    WAREHOUSE_TYPE = 'STANDARD'
    AUTO_SUSPEND = 60
    AUTO_RESUME = TRUE
    INITIALLY_SUSPENDED = TRUE;

SHOW WAREHOUSES;

-- Grant role to users
GRANT ROLE HEALTH_ROLE
TO USER JOSEPHVILLARIN27;

GRANT USAGE ON WAREHOUSE HEALTH_WH
TO ROLE HEALTH_ROLE;

GRANT USAGE ON DATABASE HEALTH_DB
TO ROLE HEALTH_ROLE;

GRANT USAGE ON SCHEMA HEALTH_DB.HEALTH_SCHEMA
TO ROLE HEALTH_ROLE;

GRANT INSERT ON TABLE HEALTH_DB.HEALTH_SCHEMA.PATIENT_EVENTS_STREAM
TO ROLE HEALTH_ROLE;

GRANT CREATE AUTHENTICATION POLICY
ON HEALTH_DB.HEALTH_SCHEMA
TO ROLE HEALTH_ROLE;

SHOW USERS;
SHOW PIPES;
USE ROLE ACCOUNTADMIN

--Create DB and Schema
CREATE OR REPLACE DATABASE HEALTH_DB;
CREATE OR REPLACE SCHEMA HEALTH_DB.HEALTH_SCHEMA;

ALTER USER JOSEPHVILLARIN27 SET RSA_PUBLIC_KEY='insert rsa key here';

-- Configure authentication policy (Optional, but recommended for explicit control)
-- NOTE: Must run as ACCOUNTADMIN since CREATE AUTHENTICATION POLICY requires schema-level privilege
CREATE OR REPLACE AUTHENTICATION POLICY HEALTH_DB.HEALTH_SCHEMA.testing_auth_policy
  AUTHENTICATION_METHODS = ('KEYPAIR')
  CLIENT_TYPES = ('DRIVERS');

-- Apply authentication policy (if created)
ALTER USER JOSEPHVILLARIN27 SET AUTHENTICATION POLICY HEALTH_DB.HEALTH_SCHEMA.testing_auth_policy;


CREATE TABLE IF NOT EXISTS PATIENT_EVENTS_STREAM
(
    EVENT_ID STRING,
    PATIENT_ID STRING,
    DEVICE_ID STRING,
    EVENT_TYPE STRING,
    HEART_RATE NUMBER,
    BLOOD_PRESSURE VARIANT,
    TEMPERATURE FLOAT,
    OXYGEN_LEVEL NUMBER,
    STATUS STRING,
    EVENT_TIME TIMESTAMP_NTZ
);

SHOW TABLES;
