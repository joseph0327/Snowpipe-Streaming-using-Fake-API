-- Transaction stream setup: user, warehouse, database, and table creation
-- Co-authored with CoCo

USE ROLE ACCOUNTADMIN

-- Create role
CREATE ROLE IF NOT EXISTS TRANSACTION_USER_ROLE;

SHOW ROLES;

-- Create the Warehouse
CREATE OR REPLACE WAREHOUSE TRANSACTION_WH
    WAREHOUSE_SIZE = 'XSMALL'
    WAREHOUSE_TYPE = 'STANDARD'
    AUTO_SUSPEND = 60
    AUTO_RESUME = TRUE
    INITIALLY_SUSPENDED = TRUE;

SHOW WAREHOUSES;


-- Grant role to users
GRANT ROLE TRANSACTION_USER_ROLE
TO USER JOSEPHVILLARIN27;

GRANT USAGE ON WAREHOUSE TRANSACTION_WH
TO ROLE TRANSACTION_USER_ROLE;

GRANT USAGE ON DATABASE TRANSACTIONSBD
TO ROLE TRANSACTION_USER_ROLE;

GRANT USAGE ON SCHEMA TRANSACTIONSBD.TRANSACTION_SCHEMA
TO ROLE TRANSACTION_USER_ROLE;

GRANT INSERT ON TABLE TRANSACTIONSBD.TRANSACTION_SCHEMA.TRANSACTIONS_TABLE_STREAMING
TO ROLE TRANSACTION_USER_ROLE;

GRANT CREATE AUTHENTICATION POLICY
ON SCHEMA TRANSACTIONSBD.TRANSACTION_SCHEMA
TO ROLE TRANSACTION_USER_ROLE;

SHOW USERS;

-- Create database and schema
CREATE OR REPLACE DATABASE TRANSACTIONSBD;
CREATE OR REPLACE SCHEMA TRANSACTIONSBD.transaction_schema;

-- Create a target table
CREATE OR REPLACE TABLE  TRANSACTIONSBD.transaction_schema.transactions_table(
    transaction_id VARCHAR NOT NULL,
    sequence NUMBER NOT NULL,
    merchant VARCHAR,
    amount FLOAT,
    currency VARCHAR(10),
    status  VARCHAR(10),
    ts TIMESTAMP_NTZ
);

SELECT * FROM TRANSACTIONSBD.transaction_schema.transactions_table;

-- Configure authentication policy (Optional, but recommended for explicit control)
-- NOTE: Must run as ACCOUNTADMIN since CREATE AUTHENTICATION POLICY requires schema-level privilege
CREATE OR REPLACE AUTHENTICATION POLICY TRANSACTIONSBD.TRANSACTION_SCHEMA.testing_auth_policy
  AUTHENTICATION_METHODS = ('KEYPAIR')
  CLIENT_TYPES = ('DRIVERS');

-- Apply authentication policy (if created)
ALTER USER JOSEPHVILLARIN27 SET AUTHENTICATION POLICY TRANSACTIONSBD.TRANSACTION_SCHEMA.testing_auth_policy;


-- Set the public key for key-pair authentication
-- NOTE: Replace 'YOUR_FORMATTED_PUBLIC_KEY' with the output of the PUBK variable from the key generation step.
ALTER USER JOSEPHVILLARIN27 SET RSA_PUBLIC_KEY='Insert RSA key';


SHOW CHANNELS;

SHOW PIPES;

SHOW CHANNELS IN SCHEMA TRANSACTIONSBD.TRANSACTION_SCHEMA;



CREATE TABLE IF NOT EXISTS TRANSACTIONSBD.TRANSACTION_SCHEMA.TRANSACTIONS_TABLE_STREAMING (
    transaction_id STRING NOT NULL,
    sequence NUMBER,
    source STRING,
    customer_id STRING,
    account_id STRING,
    merchant STRING,
    transaction_type STRING,
    amount NUMBER(10,2),
    currency STRING,
    status STRING,
    channel STRING,
    country STRING,
    device_id STRING,
    reference_number STRING,
    event_time TIMESTAMP_NTZ
);

SELECT * FROM TRANSACTIONSBD.TRANSACTION_SCHEMA.TRANSACTIONS_TABLE_STREAMING;
